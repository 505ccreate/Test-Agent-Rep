  function ensureCtx() {
    if (!ctx) {
      ctx = new (window.AudioContext || window.webkitAudioContext)();

  // ---------- Noise ----------
  function makeNoiseBuffer(type) {
    const len = ctx.sampleRate * 4;
    const buf = ctx.createBuffer(1, len, ctx.sampleRate);
    const d = buf.getChannelData(0);
    if (type === 'white') {
      for (let i = 0; i < len; i++) d[i] = Math.random() * 2 - 1;
    } else if (type === 'pink') {
      let b0=0,b1=0,b2=0,b3=0,b4=0,b5=0,b6=0;
      for (let i = 0; i < len; i++) {
        const w = Math.random() * 2 - 1;
        b0 = 0.99886*b0 + w*0.0555179; b1 = 0.99332*b1 + w*0.0750759;
        b2 = 0.96900*b2 + w*0.1538520; b3 = 0.86650*b3 + w*0.3104856;
        b4 = 0.55000*b4 + w*0.5329522; b5 = -0.7616*b5 - w*0.0168980;
        d[i] = (b0+b1+b2+b3+b4+b5+b6 + w*0.5362) * 0.11;
        b6 = w * 0.115926;
      }
    } else {
      let last = 0;
      for (let i = 0; i < len; i++) {
        const w = Math.random() * 2 - 1;
        last = (last + 0.02 * w) / 1.02;
        d[i] = last * 3.5;
      }
    }
    return buf;
  }
  let burstBuf = null;
  function makeShortBurst() {
    if (burstBuf) return burstBuf;
    const len = Math.floor(ctx.sampleRate * 0.03);
    burstBuf = ctx.createBuffer(1, len, ctx.sampleRate);
    const d = burstBuf.getChannelData(0);
    for (let i = 0; i < len; i++) d[i] = (Math.random()*2-1) * (1 - i/len);
    return burstBuf;
  }

  function buildNoise(type, level, dest) {
    const src = ctx.createBufferSource();
    src.buffer = makeNoiseBuffer(type);
    src.loop = true;
    const g = ctx.createGain();
    g.gain.value = 0;
    src.connect(g); g.connect(dest || master);
    src.start();
    return { stops: [src], gain: g, level: level };
  }
  function buildRain(dest, heavy) {
    const base = buildNoise('pink', 1.0, dest);
    const hp = ctx.createBiquadFilter(); hp.type = 'highpass'; hp.frequency.value = heavy ? 300 : 400;
    const lp = ctx.createBiquadFilter(); lp.type = 'lowpass'; lp.frequency.value = heavy ? 9000 : 7000;
    base.stops[0].disconnect();
    base.stops[0].connect(hp); hp.connect(lp); lp.connect(base.gain);
    const dropGain = ctx.createGain(); dropGain.gain.value = heavy ? 0.5 : 0.35;
    dropGain.connect(base.gain);
    let alive = true;
    function drop() {
      if (!alive) return;
      const o = ctx.createBufferSource();
      o.buffer = makeShortBurst();
      const f = ctx.createBiquadFilter();
      f.type = 'bandpass'; f.frequency.value = 1500 + Math.random() * 5000; f.Q.value = 8;
      const g2 = ctx.createGain(); g2.gain.value = 0.10 + Math.random() * 0.20;
      o.connect(f); f.connect(g2); g2.connect(dropGain);
      o.start();
      setTimeout(drop, heavy ? (15 + Math.random() * 80) : (30 + Math.random() * 160));
    }
    drop();
    base.stopExtra = () => { alive = false; };
    return base;
  }
  function buildOcean() {
    const base = buildNoise('brown', 1.0);
    const lp = ctx.createBiquadFilter(); lp.type = 'lowpass'; lp.frequency.value = 850;
    base.stops[0].disconnect();
    base.stops[0].connect(lp);
    const swell = ctx.createGain(); swell.gain.value = 0.55;
    lp.connect(swell); swell.connect(base.gain);
    const lfo = ctx.createOscillator(); lfo.frequency.value = 0.09;
    const amt = ctx.createGain(); amt.gain.value = 0.4;
    lfo.connect(amt); amt.connect(swell.gain);
    lfo.start();
    base.stops.push(lfo);
    return base;
  }
  function buildStream() {
    const base = buildNoise('white', 0.8);
    const bp = ctx.createBiquadFilter(); bp.type = 'bandpass'; bp.frequency.value = 1200; bp.Q.value = 0.7;
    base.stops[0].disconnect();
    base.stops[0].connect(bp); bp.connect(base.gain);
    const lfo = ctx.createOscillator(); lfo.frequency.value = 0.5;
    const amt = ctx.createGain(); amt.gain.value = 350;
    lfo.connect(amt); amt.connect(bp.frequency);
    lfo.start();
    base.stops.push(lfo);
    return base;
  }
