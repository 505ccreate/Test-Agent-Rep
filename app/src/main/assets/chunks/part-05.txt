  function startBath() {
    if (bath.live) return;
    const panner = ctx.createPanner();
    panner.panningModel = 'HRTF';
    panner.distanceModel = 'inverse';
    panner.refDistance = 0.5;
    panner.rolloffFactor = 1.2;
    panner.connect(master);
    bath.panner = panner;
    bath.live = buildRain(panner, true);
    bath.live.gain.gain.setTargetAtTime((+bathVol.value/100) * 0.5, ctx.currentTime, 0.6);
    bath.phase = 0;
    bath.lastT = performance.now();
    orbit();
  }
  function stopBath() {
    if (!bath.live) return;
    const live = bath.live, panner = bath.panner;
    bath.live = null; bath.panner = null;
    live.gain.gain.setTargetAtTime(0, ctx.currentTime, 0.2);
    setTimeout(() => {
      live.stops.forEach(n => { try { n.stop(); } catch(e){} });
      if (live.stopExtra) live.stopExtra();
      try { panner.disconnect(); } catch(e){}
    }, 700);
  }

  // The orbit path: a continuous journey that
  //   1) circles ear-to-ear around the head (azimuth sweep),
  //   2) arcs from the nape, over the crown, to the forehead (elevation sweep),
  //   3) periodically sinks low behind/below (the "washing down" phase).
  // Three incommensurate rates so the path never exactly repeats — like a real massage.
  function orbit() {
    if (!bath.live || !bath.panner) return;
    const now = performance.now();
    const dt = Math.min((now - bath.lastT) / 1000, 0.1);
    bath.lastT = now;
    const speed = 0.15 + (+bathSpeed.value / 100) * 0.85; // rad-ish multiplier
    bath.phase += dt * speed;

    const p = bath.phase;
    const az = p * 0.9;                       // slow circle around the head
    const el = Math.sin(p * 0.37) * 1.1;      // nape -> crown -> forehead sweep
    const sink = Math.max(0, Math.sin(p * 0.13)) * 1.4; // occasional wash downward
    const r = 1.0;

    const x = r * Math.cos(el) * Math.sin(az);          // left-right
    const y = r * Math.sin(el) - sink;                  // over crown / down the neck
    const z = -r * Math.cos(el) * Math.cos(az);         // front-back

    const t = ctx.currentTime;
    if (bath.panner.positionX) {
      bath.panner.positionX.setTargetAtTime(x, t, 0.05);
      bath.panner.positionY.setTargetAtTime(y, t, 0.05);
      bath.panner.positionZ.setTargetAtTime(z, t, 0.05);
    } else {
      bath.panner.setPosition(x, y, z);
    }
    requestAnimationFrame(orbit);
  }
